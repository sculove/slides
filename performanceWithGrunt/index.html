Gruntfile.js

  • Depth1 : Task, property
  • Depth2 : Target, options
grunt.initConfig({
	// task
	task : {
		// option of all target
		options : {},

		// target
		foo : {
			// option of target
			options : {}
			// src와 dest
			src : [ "src/**/*.js", "src/**/*.json" ],
			dest : "dest/dest.js"
		},
		bar : {
			 "dest" : "source"
		},
		zoo : {
	     				files: [{
						expand: true,
						src: "<%= task.foo.dest %>",
						dest: "<%= pkg.workspace %>/css"
			}]
		}
	},
	// property
	pkg : grunt.file.readJSON("package.json")
});
grunt.loadNpmTasks("task"); 	// load task
grunt.registerTask("default", []);	// register task
